void Pop(int *a, int n) {
	int swap;
	for (int i = 0; i < n; i++) {
		for (int j = 0; j < i ; j++) {
			if (a[i] < a[j]) {
				swap = a[i];
				a[i] = a[j];
				a[j] = swap;
			}
		}
	}
}

int BinSearch(int * a, int n, int x) {
	int low = 0;
	int high = n - 1;
	int mid = 0;
	while (low <= high) {
		mid = ((low + high) >> 1);
		if(a[mid] > x){
			high = mid - 1;
		} else if(a[mid] < x){
			low = mid + 1 ;
		} else {
			printf("In Bin Search: Index:%d\n",mid);
			return mid;
		} 
	}
	return -1;
}
