                                                      • з§ҒжңүеұһжҖ§ е°ұжҳҜ еҜ№иұЎ дёҚеёҢжңӣе…¬ејҖзҡ„ еұһжҖ§
                                                      • з§Ғжңүж–№жі• е°ұжҳҜ еҜ№иұЎ дёҚеёҢжңӣе…¬ејҖзҡ„ ж–№жі•

                                                      class Women:

    def __init__(self, name):

        self.name = name
        # дёҚиҰҒй—®еҘіз”ҹзҡ„е№ҙйҫ„
        self.__age = 18

    def __secret(self):
        print("жҲ‘зҡ„е№ҙйҫ„жҳҜ %d" % self.__age)
