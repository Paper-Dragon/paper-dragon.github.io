  1. networkpolicyеҺҹзҗҶ

еңЁе…·дҪ“е®һзҺ°дёҠпјҢеҮЎжҳҜж”ҜжҢҒ NetworkPolicy зҡ„ CNI зҪ‘з»ңжҸ’д»¶пјҢйғҪз»ҙжҠӨзқҖдёҖдёӘ NetworkPolicy ControllerпјҢйҖҡиҝҮжҺ§еҲ¶еҫӘзҺҜзҡ„ж–№ејҸеҜ№ NetworkPolicy еҜ№иұЎзҡ„еўһеҲ ж”№жҹҘеҒҡеҮәе“Қеә”пјҢз„¶еҗҺеңЁе®ҝдё»жңәдёҠе®ҢжҲҗ iptables и§„еҲҷзҡ„й…ҚзҪ®е·ҘдҪңгҖӮ

еңЁ Kubernetes з”ҹжҖҒйҮҢпјҢзӣ®еүҚе·Із»Ҹе®һзҺ°дәҶ NetworkPolicy зҡ„зҪ‘з»ңжҸ’д»¶еҢ…жӢ¬ CalicoгҖҒWeave е’Ң kube-router зӯүеӨҡдёӘйЎ№зӣ®пјҢдҪҶжҳҜе№¶дёҚеҢ…жӢ¬ Flannel йЎ№зӣ®гҖӮжүҖд»ҘиҜҙпјҢеҰӮжһңжғіиҰҒеңЁдҪҝз”Ё Flannel зҡ„еҗҢж—¶иҝҳдҪҝз”Ё NetworkPolicy зҡ„иҜқпјҢдҪ е°ұйңҖиҰҒеҶҚйўқеӨ–е®үиЈ…дёҖдёӘзҪ‘з»ңжҸ’д»¶пјҢжҜ”еҰӮ Calico йЎ№зӣ®пјҢжқҘиҙҹиҙЈжү§иЎҢ NetworkPolicyгҖӮ

е®үиЈ… Flannel + Calico зҡ„жөҒзЁӢйқһеёёз®ҖеҚ•пјҢдҪ зӣҙжҺҘеҸӮиҖғиҝҷдёӘж–ҮжЎЈдёҖй”®е®үиЈ…еҚіеҸҜгҖӮ

йӮЈд№ҲпјҢиҝҷдәӣзҪ‘з»ңжҸ’д»¶пјҢеҸҲжҳҜеҰӮдҪ•ж №жҚ® NetworkPolicy еҜ№ Pod иҝӣиЎҢйҡ”зҰ»зҡ„е‘ўпјҹ

жҺҘдёӢжқҘпјҢжҲ‘е°ұд»ҘдёүеұӮзҪ‘з»ңжҸ’д»¶дёәдҫӢпјҲжҜ”еҰӮ Calico е’Ң kube-routerпјүпјҢжқҘдёәдҪ еҲҶжһҗдёҖдёӢиҝҷйғЁеҲҶзҡ„еҺҹзҗҶгҖӮдёәдәҶж–№дҫҝи®Іи§ЈпјҢиҝҷдёҖж¬ЎжҲ‘зј–еҶҷдәҶдёҖдёӘжҜ”иҫғз®ҖеҚ•зҡ„ NetworkPolicy еҜ№иұЎпјҢеҰӮдёӢжүҖзӨәпјҡ

apiVersion: extensions/v1beta1
kind: NetworkPolicy
metadata:
  name: test-network-policy
  namespace: default
spec:
  podSelector:
    matchLabels:
      role: db
  ingress:
   - from:
     - namespaceSelector:
         matchLabels:
           project: myproject
     - podSelector:
         matchLabels:
           role: frontend
       ports:
       - protocol: tcp
         port: 6379

еҸҜд»ҘзңӢеҲ°пјҢжҲ‘д»¬жҢҮе®ҡзҡ„ ingressвҖңзҷҪеҗҚеҚ•вҖқпјҢжҳҜд»»дҪ• Namespace йҮҢпјҢжҗәеёҰ project=myproject ж Үзӯҫзҡ„ Namespace йҮҢзҡ„ Podпјӣд»ҘеҸҠ default Namespace йҮҢпјҢжҗәеёҰдәҶ role=frontend ж Үзӯҫзҡ„ PodгҖӮе…Ғи®ёиў«и®ҝй—®зҡ„з«ҜеҸЈжҳҜпјҡ6379гҖӮ

иҖҢиў«йҡ”зҰ»зҡ„еҜ№иұЎпјҢжҳҜжүҖжңүжҗәеёҰдәҶ role=db ж Үзӯҫзҡ„ PodгҖӮ

йӮЈд№ҲиҝҷдёӘж—¶еҖҷпјҢKubernetes зҡ„зҪ‘з»ңжҸ’д»¶е°ұдјҡдҪҝз”ЁиҝҷдёӘ NetworkPolicy зҡ„е®ҡд№үпјҢеңЁе®ҝдё»жңәдёҠз”ҹжҲҗ iptables и§„еҲҷгҖӮиҝҷдёӘиҝҮзЁӢпјҢжҲ‘еҸҜд»ҘйҖҡиҝҮеҰӮдёӢжүҖзӨәзҡ„дёҖж®ө Go иҜӯиЁҖйЈҺж јзҡ„дјӘд»Јз ҒжқҘдёәдҪ жҸҸиҝ°пјҡ

for dstIP := range жүҖжңүиў«networkpolicy.spec.podSelectorйҖүдёӯзҡ„Podзҡ„IPең°еқҖ
  for srcIP := range жүҖжңүиў«ingress.from.podSelectorйҖүдёӯзҡ„Podзҡ„IPең°еқҖ
    for port, protocol := range ingress.ports {
      iptables -A KUBE-NWPLCY-CHAIN -s $srcIP -d $dstIP -p $protocol -m $protocol --dport $port -j ACCEPT 
    }
  }
}

еҸҜд»ҘзңӢеҲ°пјҢиҝҷжҳҜдёҖжқЎжңҖеҹәжң¬зҡ„гҖҒйҖҡиҝҮеҢ№й…ҚжқЎд»¶еҶіе®ҡдёӢдёҖжӯҘеҠЁдҪңзҡ„ iptables и§„еҲҷгҖӮиҝҷжқЎи§„еҲҷзҡ„еҗҚеӯ—жҳҜ KUBE-NWPLCY-CHAINпјҢеҗ«д№үжҳҜпјҡеҪ“ IP еҢ…зҡ„жәҗең°еқҖжҳҜ srcIPгҖҒзӣ®зҡ„ең°еқҖжҳҜ dstIPгҖҒеҚҸи®®жҳҜ protocolгҖҒзӣ®зҡ„з«ҜеҸЈжҳҜ port зҡ„ж—¶еҖҷпјҢе°ұе…Ғи®ёе®ғйҖҡиҝҮпјҲACCEPTпјүгҖӮ

иҖҢжӯЈеҰӮиҝҷж®өдјӘд»Јз ҒжүҖзӨәпјҢеҢ№й…ҚиҝҷжқЎи§„еҲҷжүҖйңҖзҡ„иҝҷеӣӣдёӘеҸӮж•°пјҢйғҪжҳҜд»Һ NetworkPolicy еҜ№иұЎйҮҢиҜ»еҸ–еҮәжқҘзҡ„гҖӮ

еҸҜд»ҘзңӢеҲ°пјҢKubernetes зҪ‘з»ңжҸ’д»¶еҜ№ Pod иҝӣиЎҢйҡ”зҰ»пјҢе…¶е®һжҳҜйқ еңЁе®ҝдё»жңәдёҠз”ҹжҲҗ NetworkPolicy еҜ№еә”зҡ„ iptable и§„еҲҷжқҘе®һзҺ°зҡ„гҖӮ

жӯӨеӨ–пјҢеңЁи®ҫзҪ®еҘҪдёҠиҝ°вҖңйҡ”зҰ»вҖқи§„еҲҷд№ӢеҗҺпјҢзҪ‘з»ңжҸ’д»¶иҝҳйңҖиҰҒжғіеҠһжі•пјҢе°ҶжүҖжңүеҜ№иў«йҡ”зҰ» Pod зҡ„и®ҝй—®иҜ·жұӮпјҢйғҪиҪ¬еҸ‘еҲ°дёҠиҝ° KUBE-NWPLCY-CHAIN и§„еҲҷдёҠеҺ»иҝӣиЎҢеҢ№й…ҚгҖӮе№¶дё”пјҢеҰӮжһңеҢ№й…ҚдёҚйҖҡиҝҮпјҢиҝҷдёӘиҜ·жұӮеә”иҜҘиў«вҖңжӢ’з»қвҖқгҖӮ

еңЁ CNI зҪ‘з»ңжҸ’д»¶дёӯпјҢдёҠиҝ°йңҖжұӮеҸҜд»ҘйҖҡиҝҮи®ҫзҪ®дёӨз»„ iptables и§„еҲҷжқҘе®һзҺ°гҖӮ

з¬¬дёҖз»„и§„еҲҷпјҢиҙҹиҙЈвҖңжӢҰжҲӘвҖқеҜ№иў«йҡ”зҰ» Pod зҡ„и®ҝй—®иҜ·жұӮгҖӮз”ҹжҲҗиҝҷдёҖз»„и§„еҲҷзҡ„дјӘд»Јз ҒпјҢеҰӮдёӢжүҖзӨәпјҡ

for pod := range иҜҘNodeдёҠзҡ„жүҖжңүPod {
    if podжҳҜnetworkpolicy.spec.podSelectorйҖүдёӯзҡ„ {
        iptables -A FORWARD -d $podIP -m physdev --physdev-is-bridged -j KUBE-POD-SPECIFIC-FW-CHAIN
        iptables -A FORWARD -d $podIP -j KUBE-POD-SPECIFIC-FW-CHAIN
        ...
    }
}

Kubernetes зҡ„зҪ‘з»ңжЁЎеһӢд»ҘеҸҠеӨ§еӨҡж•°е®№еҷЁзҪ‘з»ңе®һзҺ°пјҢе…¶е®һж—ўдёҚдјҡдҝқиҜҒе®№еҷЁд№Ӣй—ҙдәҢеұӮзҪ‘з»ңзҡ„дә’йҖҡпјҢд№ҹдёҚдјҡе®һзҺ°е®№еҷЁд№Ӣй—ҙзҡ„дәҢеұӮзҪ‘з»ңйҡ”зҰ»гҖӮиҝҷи·ҹ IaaS йЎ№зӣ®з®ЎзҗҶиҷҡжӢҹжңәзҡ„ж–№ејҸпјҢжҳҜе®Ңе…ЁдёҚеҗҢзҡ„гҖӮжүҖд»ҘиҜҙпјҢKubernetes д»Һеә•еұӮзҡ„и®ҫи®Ўе’Ңе®һзҺ°дёҠпјҢжӣҙеҖҫеҗ‘дәҺеҒҮи®ҫдҪ е·Із»ҸжңүдәҶдёҖеҘ—е®Ңж•ҙзҡ„зү©зҗҶеҹәзЎҖи®ҫж–ҪгҖӮз„¶еҗҺпјҢKubernetes иҙҹиҙЈеңЁжӯӨеҹәзЎҖдёҠжҸҗдҫӣдёҖз§ҚвҖңејұеӨҡз§ҹжҲ·вҖқпјҲsoft multi-tenancyпјүзҡ„иғҪеҠӣгҖӮ

е№¶дё”пјҢеҹәдәҺдёҠиҝ°жҖқи·ҜпјҢKubernetes е°ҶжқҘд№ҹдёҚеӨ§еҸҜиғҪжҠҠ Namespace еҸҳжҲҗдёҖдёӘе…·жңүе®һиҙЁж„Ҹд№үзҡ„йҡ”зҰ»жңәеҲ¶пјҢжҲ–иҖ…жҠҠе®ғжҳ е°„жҲҗдёәвҖңеӯҗзҪ‘вҖқжҲ–иҖ…вҖңз§ҹжҲ·вҖқгҖӮжҜ•з«ҹдҪ еҸҜд»ҘзңӢеҲ°пјҢNetworkPolicy еҜ№иұЎзҡ„жҸҸиҝ°иғҪеҠӣпјҢиҰҒжҜ”еҹәдәҺ Namespace зҡ„еҲ’еҲҶдё°еҜҢеҫ—еӨҡгҖӮиҝҷд№ҹжҳҜдёәд»Җд№ҲпјҢеҲ°зӣ®еүҚдёәжӯўпјҢKubernetes йЎ№зӣ®еңЁдә‘и®Ўз®—з”ҹжҖҒйҮҢзҡ„е®ҡдҪҚпјҢе…¶е®һжҳҜеҹәзЎҖи®ҫж–ҪдёҺ PaaS д№Ӣй—ҙзҡ„дёӯй—ҙеұӮгҖӮиҝҷжҳҜйқһеёёз¬ҰеҗҲвҖңе®№еҷЁвҖқиҝҷдёӘжң¬иҙЁдёҠе°ұжҳҜиҝӣзЁӢзҡ„жҠҪиұЎзІ’еәҰзҡ„гҖӮеҪ“з„¶пјҢйҡҸзқҖ Kubernetes зӨҫеҢәд»ҘеҸҠ CNCF з”ҹжҖҒзҡ„дёҚж–ӯеҸ‘еұ•пјҢKubernetes йЎ№зӣ®д№ҹе·Із»ҸејҖе§ӢйҖҗжӯҘдёӢжҺўпјҢвҖңеҗғвҖқжҺүдәҶеҹәзЎҖи®ҫж–ҪйўҶеҹҹзҡ„еҫҲеӨҡвҖңиӣӢзі•вҖқгҖӮиҝҷд№ҹжӯЈжҳҜе®№еҷЁз”ҹжҖҒз»§з»ӯеҸ‘еұ•зҡ„дёҖдёӘеҝ…з„¶ж–№еҗ‘гҖӮ
